nn <- 
  last_p |> 
  mutate(
    fecha = str_replace_all(fecha, "Set", "Sep")
    , fecha1 = str_to_lower(fecha)
    , fecha2 = paste(str_sub(fecha1, 1, 5), ".", str_sub(fecha1, 6, -1), sep = "")
    , fecha = dmy(fecha2)
  ) |> 
  # filter(is.na(fecha))
  dplyr::select(1:5) |> 
  rename(
    oro = 2
    , zinc = 3
    , plata = 4
    , cobre = 5
  ) |> 
  pivot_longer(!fecha) |> 
  mutate(
    value = as.numeric(value)
  )
