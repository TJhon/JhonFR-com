sport_diff <- 
  sports |> 
  select(contains(c("rev", "exp"))) |> 
  select(!contains("total")) |> 
  gather() |> 
  separate(key, c("tipo", "sex"), sep = "_") |> 
  drop_na(value) 

sport_w <- 
  sport_diff |> 
  group_by(tipo) |> 
  mutate(id = row_number()) |> 
  pivot_wider(names_from = tipo, values_from = value) |> 
  ungroup() |> 
  select(!id) |> 
  mutate(sex = ifelse(sex == "men", "Hombres", "Mujeres"))
